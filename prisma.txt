model Task {
  id            Int        @id @default(autoincrement())
  title         String
  taskDate      DateTime
  status        TaskStatus @default(PENDING)
  pendingReason String?

  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  department   Department @relation("DepartmentTasks", fields: [departmentId], references: [id])
  departmentId Int

  currentDepartmentId Int

  // Replace single assignee fields with a JSON array
  assignees Json @default("[]")  // [{ id, name, role: "Employee"|"Manager" }]

  createdBy   Manager? @relation(fields: [createdById], references: [id])
  createdById Int?

  transfers TaskTransfer[]
}
